[sqlite3]======================================
CREATE TABLE keywords (
	id			INTEGER PRIMARY KEY     AUTOINCREMENT	NOT NULL,
	created_at	VARCHAR(30),
	updated_at	VARCHAR(30),
	
    word		VARCHAR(30),
    
    rubi		VARCHAR(50),
    
    genre_id	INTEGER,
    
    type_id		INTEGER,

    memo		VARCHAR(50)
	
);

DROP TABLE keywords;
DROP TABLE keywords_old;

		id, created_at, updated_at, word, genre_id, type_id, memo
